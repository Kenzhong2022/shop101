# 用户头像上传实现文档

## 概述

本文档介绍如何在 Nuxt3 项目中实现用户头像上传功能，使用 Cloudinary 作为图片存储服务。

## 技术栈

- **前端框架**: Nuxt3 + Vue3
- **UI 组件库**: Element Plus
- **图片存储**: Cloudinary
- **状态管理**: Pinia (useUser composable)

## 现有实现分析

### 1. 核心 composable: `useCloudinary.js`

位置: `app/composables/useCloudinary.js`

```javascript
import { ref } from "vue";

const CLOUD_NAME = "dlji1nmdj";
const UPLOAD_PRESET = "shop101-upload-preset";

/**
 * 上传图片到 Cloudinary
 * @param {boolean} immediateUpload - 是否选择后立即上传
 * @returns {Object} 上传相关状态和函数
 */
export function useCloudinary(immediateUpload = false) {
  const file = ref(null);           // 原始 File 对象
  const previewUrl = ref("");       // 本地预览地址
  const resultUrl = ref("");        // 上传后的 URL
  const uploading = ref(false);     // 上传状态

  // 选择文件
  function selectFile(rawFile) {
    if (!rawFile) return;
    file.value = rawFile;
    previewUrl.value = URL.createObjectURL(rawFile);
    resultUrl.value = "";
    if (immediateUpload) {
      upload();
    }
    return previewUrl.value;
  }

  // 上传文件
  async function upload() {
    if (!file.value) throw new Error("请先选择图片");

    uploading.value = true;
    const nameNoExt = file.value.name.replace(/\.[^.]+$/, "");
    const fd = new FormData();
    fd.append("file", file.value);
    fd.append("upload_preset", UPLOAD_PRESET);
    fd.append("public_id", nameNoExt);

    try {
      const res = await fetch(
        `https://api.cloudinary.com/v1_1/${CLOUD_NAME}/image/upload`,
        { method: "POST", body: fd }
      );
      const data = await res.json();
      resultUrl.value = data.secure_url;
      ElMessage.success("上传成功");
      return data.secure_url;
    } finally {
      uploading.value = false;
    }
  }

  // 重置状态
  function reset() {
    if (previewUrl.value.startsWith("blob:"))
      URL.revokeObjectURL(previewUrl.value);
    file.value = null;
    previewUrl.value = "";
    resultUrl.value = "";
  }

  return {
    file,
    previewUrl,
    resultUrl,
    uploading,
    selectFile,
    upload,
    reset,
  };
}
```

```typescript
<script setup lang="ts">
import { useUser } from "~/composables/useUser";

// 用户状态
const userState = useUser();

// 头像悬停状态
const isHovered = ref(false);

// 文件输入引用
const fileInput = ref<HTMLInputElement>();

// 使用 Cloudinary composable，immediateUpload=true 表示选择后立即上传
const { file, previewUrl, resultUrl, uploading, selectFile, upload } =
  useCloudinary(true);

// 监听上传结果，成功后更新用户头像
watch(resultUrl, async (newUrl) => {
  if (newUrl) {
    // 调用 API 更新用户头像
    await updateUserAvatar(newUrl);
    // 更新本地状态
    userState.userInfo.avatar = newUrl;
    ElMessage.success("头像更新成功");
  }
});

// 处理文件选择
async function onSelect(e: Event) {
  const target = e.target as HTMLInputElement;
  const selectedFile = target.files?.[0];
  
  if (!selectedFile) return;
  
  // 验证文件类型
  if (!selectedFile.type.startsWith('image/')) {
    ElMessage.error("请选择图片文件");
    return;
  }
  
  // 验证文件大小 (最大 2MB)
  const maxSize = 2 * 1024 * 1024;
  if (selectedFile.size > maxSize) {
    ElMessage.error("图片大小不能超过 2MB");
    return;
  }
  
  selectFile(selectedFile);
}

// 更新用户头像 API
async function updateUserAvatar(avatarUrl: string) {
  // 调用后端 API 更新头像
  const { data } = await $fetch('/api/user/update-avatar', {
    method: 'POST',
    body: { avatar: avatarUrl }
  });
  return data;
}
</script>
```

### 步骤 2: 创建用户头像更新 API

创建文件: `server/api/user/update-avatar.post.ts`

```typescript
import { defineEventHandler, readBody, createError } from "h3";
import { verifyToken } from "~~/server/utils/auth";
import getNeon from "~~/server/utils/neon";

export default defineEventHandler(async (event) => {
  try {
    // 1. 验证用户登录
    const token = getHeader(event, "authorization")?.replace("Bearer ", "");
    if (!token) {
      throw createError({ statusCode: 401, message: "未登录" });
    }

    const decoded = verifyToken(token);
    if (!decoded) {
      throw createError({ statusCode: 401, message: "登录已过期" });
    }

    // 2. 获取请求体
    const { avatar } = await readBody(event);
    if (!avatar) {
      throw createError({ statusCode: 400, message: "头像地址不能为空" });
    }

    // 3. 更新数据库
    const sql = getNeon();
    await sql`
      UPDATE users 
      SET avatar = ${avatar}, updated_at = NOW()
      WHERE id = ${decoded.userId}
    `;

    return {
      success: true,
      message: "头像更新成功",
      data: { avatar }
    };
  } catch (error: any) {
    console.error("更新头像失败:", error);
    throw createError({
      statusCode: 500,
      message: error.message || "更新头像失败"
    });
  }
});
```

### 步骤 3: 更新 useUser composable

确保 `useUser` composable 有更新头像的方法:

```typescript
// composables/useUser.ts
export const useUser = () => {
  const userInfo = useState<UserInfo>("userInfo", () => ({
    // ... 其他字段
    avatar: "",
  }));

  // 更新头像
  async function updateAvatar(avatarUrl: string) {
    try {
      const { data } = await $fetch("/api/user/update-avatar", {
        method: "POST",
        body: { avatar: avatarUrl },
      });
      
      // 更新本地状态
      userInfo.value.avatar = avatarUrl;
      
      // 更新 localStorage
      const stored = localStorage.getItem("userInfo");
      if (stored) {
        const parsed = JSON.parse(stored);
        parsed.avatar = avatarUrl;
        localStorage.setItem("userInfo", JSON.stringify(parsed));
      }
      
      return data;
    } catch (error) {
      console.error("更新头像失败:", error);
      throw error;
    }
  }

  return {
    userInfo,
    updateAvatar,
    // ... 其他方法
  };
};
```

```vue
<template>

<script setup lang="ts">

    try {

      ElMessage.error("头像更新失败，请重试");

      reset();
    }
  }

// 点击头像
function handleAvatarClick() {
  if (uploading.value) return;
  fileInput.value?.click();
}

// 文件选择处理
async function onFileSelect(e: Event) {
  const target = e.target as HTMLInputElement;
  const file = target.files?.[0];
  
  if (!file) return;
  
  // 验证文件类型
  const allowedTypes = ['image/jpeg', 'image/png', 'image/webp'];
  if (!allowedTypes.includes(file.type)) {
    ElMessage.error("只支持 JPG、PNG、WebP 格式的图片");
    target.value = '';
    return;
  }
  
  // 验证文件大小 (最大 2MB)
  const maxSize = 2 * 1024 * 1024;
  if (file.size > maxSize) {
    ElMessage.error("图片大小不能超过 2MB");
    target.value = '';
    return;
  }
  
  try {
    // 选择文件并上传
    selectFile(file);
    // 手动触发上传
    const { upload } = useCloudinary(false);
    await upload();
  } catch (error) {
    console.error("上传失败:", error);
    ElMessage.error("上传失败，请重试");
  }
  
  // 清空 input，允许重复选择同一文件
  target.value = '';
}
</script>
```

## 图片压缩优化

在上传前添加图片压缩:

```typescript
import imageCompression from 'browser-image-compression';

// 压缩图片
async function compressImage(file: File): Promise<File> {
  const options = {
    maxSizeMB: 1,           // 最大 1MB
    maxWidthOrHeight: 800,  // 最大尺寸 800px
    useWebWorker: true,
    fileType: 'image/webp', // 转换为 webp
  };
  
  try {
    const compressedFile = await imageCompression(file, options);
    return compressedFile;
  } catch (error) {
    console.error('压缩失败:', error);
    return file; // 压缩失败返回原文件
  }
}

// 在 onFileSelect 中使用
async function onFileSelect(e: Event) {
  // ... 验证代码
  
  const compressedFile = await compressImage(file);
  selectFile(compressedFile);
  // ... 上传代码
}
```

## 注意事项

1. **文件验证**: 始终在前端和后端都进行文件类型和大小验证
2. **错误处理**: 提供友好的错误提示
3. **加载状态**: 上传过程中显示加载状态，防止重复提交
4. **图片压缩**: 大图片先压缩再上传，减少带宽和存储
5. **安全性**: 使用 upload preset 限制上传格式和大小
6. **备份**: 考虑保留旧头像或支持头像历史
