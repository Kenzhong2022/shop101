### 2. 列表查询接口

**文件**: `server/api/history/products/list.post.ts`

```typescript
import { defineEventHandler, readBody } from "h3";
import { requireAuth } from "~~/server/utils/auth";
import getNeon from "~~/server/utils/neon";

const mySql = getNeon();

export default defineEventHandler(async (event) => {
  // 1. JWT认证获取用户ID
  const { userId } = event.context.user;

  // 2. 获取请求参数
  const body = await readBody(event);
  const { page = 1, page_size = 10, action_type } = body;

  // 3. 查询数据：通过子查询获取每个商品的最新记录
  const result = await mySql`
    SELECT 
        id, goods_name, image, price, stock, sort, 
        is_show, average_rating, created_at_fmt, shop_name
    FROM (
        SELECT DISTINCT ON (hg.id)
            hg.id,
            hg.goods_name,
            hg.image,
            hg.price,
            hg.stock,
            hg.sort,
            hg.is_show,
            hg.average_rating,
            upb.action_time,
            TO_CHAR(upb.action_time AT TIME ZONE 'UTC' AT TIME ZONE 'Asia/Shanghai', 'YYYY-MM-DD HH24:MI:SS') as created_at_fmt,
            hg.shop_name
        FROM user_item_action upb
        INNER JOIN goods hg ON upb.item_id = hg.id
        WHERE upb.user_id = ${userId}
            ${action_type ? mySql`AND upb.action_type = ${action_type}` : mySql``}
        ORDER BY hg.id, upb.action_time DESC
    ) as sub
    ORDER BY sub.action_time DESC
    LIMIT ${page_size} OFFSET ${(page - 1) * page_size};
  `;

  return {
    code: 200,
    msg: "成功",
    data: {
      total: result.length,
      GoodsItems: result,
    },
  };
});
```

**核心逻辑说明**:

- 使用 `DISTINCT ON` 确保每个商品只显示最新记录
- 按行为时间倒序排列，最新的排在前面
- 支持按行为类型筛选

### 3. 删除记录接口

**文件**: `server/api/history/products/delete.post.ts`

```typescript
import { defineEventHandler, readBody, createError } from "h3";
import { getCookie } from "h3";
import getNeon from "~~/server/utils/neon";

const mySql = getNeon();

export default defineEventHandler(async (event) => {
  try {
    // 1. 获取请求参数
    const body = await readBody(event);
    const { productIds, action_type } = body;

    // 2. 参数验证
    if (!productIds || !Array.isArray(productIds) || productIds.length === 0) {
      throw createError({ statusCode: 400, message: "商品ID数组不能为空" });
    }

    // 3. 获取用户ID（从Cookie）
    const token = getCookie(event, "auth-token");
    const userId = token ? parseInt(token.split(".")[0]) : null;

    if (!userId) {
      throw createError({ statusCode: 401, message: "用户未登录" });
    }

    // 4. 执行删除
    const deleteResult = await mySql`
      DELETE FROM user_item_action 
      WHERE user_id = ${userId} 
        AND item_id = ANY(${productIds})
        ${action_type ? mySql`AND action_type = ${action_type}` : mySql``}
      RETURNING *
    `;

    return {
      code: 200,
      msg: "删除成功",
      data: {
        deletedCount: deleteResult.length || 0,
      },
    };
  } catch (error) {
    console.error("删除失败:", error);
    throw createError({ statusCode: 500, message: "删除失败" });
  }
});
```

  data: { type: Array, default: () => [] },

    item.isChecked = !val;

    $message.warning("请至少选择一件商品");

    await ElMessageBox.confirm(`确定删除 ${total_count.value} 件商品吗？`);

    // 用户取消

### 3. 历史浏览页面

**文件**: `app/pages/user/myHistory.vue`

```vue
<template>
  <div class="flex flex-col">
    <!-- 标题栏 -->
    <div class="text-20px font-bold flex justify-between items-center">
      <div>我的历史</div>
      <div class="flex flex-row gap-12px">
        <el-checkbox :model-value="all_check" size="large" @change="toggleAll">
          <span class="text-20px font-bold">全选</span>
          <span class="text-14px text-gray-500"
            >({{ total_count }} 件商品)</span
          >
        </el-checkbox>
        <el-button size="large" type="primary" @click="deleteSelected">
          批量删除
        </el-button>
      </div>
    </div>

    <!-- 按日期分组的历史记录 -->
    <div v-for="date in Object.keys(historyList)" :key="date">
      <div class="text-20px font-bold">
        <span>{{ date }}</span>
        <span v-if="date === today"> 今天</span>
        <span v-else-if="date === yesterday"> 昨天</span>
        <span class="text-14px text-gray-500"
          >({{ historyList[date]?.length }}件)</span
        >
      </div>
      <div class="flex flex-wrap px-20px">
        <kk-goods-item
          :data="historyList[date] || []"
          :collected-mode="true"
          @click="handleClick"
          @check-change="({ id, checked }) => updateHistoryItem(id, checked)"
        />
      </div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, watch } from "vue";
import dayjs from "dayjs";
import { userBehaviorProductsDelete } from "~/api/user-behavior/products";

const props = defineProps({
  data: { type: Array, default: () => [] },
  isLoading: { type: Boolean, required: true },
});

const emit = defineEmits(["refresh"]);
const { $message } = useNuxtApp();

// 按日期分组的历史记录
const historyList = ref({});

// 计算今天和昨天的日期
const today = dayjs().format("YYYY-MM-DD");
const yesterday = dayjs().subtract(1, "day").format("YYYY-MM-DD");

// 监听数据变化，按日期分组
watch(
  () => props.data,
  (newData) => {
    historyList.value = {};
    if (Array.isArray(newData)) {
      newData.forEach((item) => {
        const date = item.created_at_fmt?.split(" ")[0] || "未知日期";
        if (!historyList.value[date]) historyList.value[date] = [];
        historyList.value[date].push({ ...item, isChecked: false });
      });
    }
  },
  { immediate: true },
);

// 全选状态
const all_check = computed(() => {
  for (const date in historyList.value) {
    if (historyList.value[date]?.some((item) => !item.isChecked)) return false;
  }
  return true;
});

// 选中的商品ID
const selected_ids = computed(() => {
  const ids = [];
  for (const date in historyList.value) {
    historyList.value[date]?.forEach((item) => {
      if (item.isChecked) ids.push(item.id);
    });
  }
  return ids;
});

// 选中数量
const total_count = computed(() => {
  let count = 0;
  for (const date in historyList.value) {
    count +=
      historyList.value[date]?.filter((item) => item.isChecked).length || 0;
  }
  return count;
});

// 全选/取消全选
function toggleAll(value) {
  for (const date in historyList.value) {
    historyList.value[date]?.forEach((item) => {
      item.isChecked = value;
    });
  }
}

// 批量删除
async function deleteSelected() {
  if (!total_count.value) {
    $message.warning("请至少选择一件商品");
    return;
  }

  try {
    await ElMessageBox.confirm(`确定删除 ${total_count.value} 件商品吗？`);
    await userBehaviorProductsDelete({
      productIds: selected_ids.value,
      action_type: 1, // 点击（浏览）
    });
    emit("refresh");
    $message.success(`成功删除 ${total_count.value} 件商品`);
  } catch {
    // 用户取消
  }
}

// 点击商品
function handleClick(goodsItem) {
  navigateTo(`/goods/${goodsItem.id}/detail`);
}

// 更新历史记录选中状态
function updateHistoryItem(id, checked) {
  for (const date in historyList.value) {
    const item = historyList.value[date]?.find((item) => item.id === id);
    if (item) {
      item.isChecked = checked;
      break;
    }
  }
}
</script>
```

## 行为追踪机制

### 商品行为上报

**文件**: `app/composables/useProductBehavior.ts`

```typescript
import { ref } from "vue";

/**
 * 商品行为追踪 composable
 * @param goodsId 商品ID
 * @param options 配置选项
 */
export function useProductBehavior(goodsId, options = {}) {
  const {
    behaviorType = "click",
    autoTrack = false,
    sourcePage = "",
  } = options;

  // 行为类型映射
  const behaviorMap = {
    click: 1,
    collect: 2,
    cart: 3,
    buy: 4,
  };

  /**
   * 上报行为
   */
  async function track() {
    const actionType = behaviorMap[behaviorType];
    if (!actionType) return;

    try {
      await $fetch("/api/user-behavior/track", {
        method: "POST",
        body: {
          item_id: goodsId,
          action_type: actionType,
          source_page: sourcePage,
        },
      });
    } catch (error) {
      console.error("行为上报失败:", error);
    }
  }

  // 自动追踪
  if (autoTrack) {
    track();
  }

  return {
    track,
  };
}
```

## 关键技术点

### 1. 数据去重策略

使用 PostgreSQL 的 `DISTINCT ON` 语法确保每个商品只显示最新的行为记录：

```sql
SELECT DISTINCT ON (hg.id)
    hg.id,
    hg.goods_name,
    upb.action_time
FROM user_item_action upb
INNER JOIN goods hg ON upb.item_id = hg.id
WHERE upb.user_id = ${userId}
ORDER BY hg.id, upb.action_time DESC
```

### 2. 时间格式化

将 UTC 时间转换为北京时间：

```sql
TO_CHAR(upb.action_time AT TIME ZONE 'UTC' AT TIME ZONE 'Asia/Shanghai', 'YYYY-MM-DD HH24:MI:SS') as created_at_fmt
```

### 3. 批量删除优化

使用 PostgreSQL 的 `ANY` 操作符高效删除多条记录：

```sql
DELETE FROM user_item_action
WHERE item_id = ANY(${productIds})
```

### 4. 权限验证

- 使用 JWT Token 进行用户认证
- 支持从 Cookie 中提取用户ID
- 确保用户只能访问和操作自己的数据

## 总结

收藏与历史浏览功能的核心是：

1. **数据模型**：基于 `user_item_action` 表记录用户行为
2. **行为类型**：通过 `action_type` 区分点击(1)、收藏(2)、加购(3)、购买(4)
3. **展示策略**：收藏和历史都只显示每个商品的最新记录
4. **交互功能**：支持单选、全选、批量删除

该实现已满足基本需求，可根据业务需要扩展搜索、时间筛选等功能。
