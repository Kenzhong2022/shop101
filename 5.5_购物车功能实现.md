# 5.5 购物车功能实现

## 5.5.1 功能概述

购物车是电商平台的核心功能之一，本项目实现了完整的购物车功能，包括：

- **商品管理**：添加、删除、数量调整
- **选中状态**：单个选中、店铺全选、全部全选
- **价格计算**：实时计算选中商品总价
- **数据持久化**：后端数据库存储，支持多端同步

## 5.5.4 后端 API 实现

### 5.5.4.1 JWT 认证机制（新增）

购物车所有接口均使用标准 JWT 认证，通过 `requireAuth()` 中间件验证用户身份。

**认证流程**：

```
客户端请求 → 获取Cookie中的auth-token → 验证JWT签名 → 返回用户ID
```

**核心认证函数**：

```typescript
// server/utils/auth.ts
export async function requireAuth(
  event: H3Event,
): Promise<{ userId: number; payload: JWTPayload }> {
  // 1. 从Cookie获取token
  const token = getCookie(event, "auth-token");

  if (!token) {
    throw createError({ statusCode: 401, message: "用户未登录" });
  }

  // 2. 验证JWT签名、过期时间、issuer、audience
  const payload = verifyJWT(token);

  if (!payload) {
    throw createError({ statusCode: 401, message: "Token无效或已过期" });
  }

  // 3. 返回用户ID和完整payload
  return { userId: payload.userId, payload };
}
```

**JWT Payload 结构**（解码后）：

```json
{
  "userId": 1,
  "email": "user@shop101.com",
  "username": "user",
  "iss": "shop101-api",
  "aud": "shop101-client",
  "iat": 1777331751,
  "exp": 1777331811
}
```

### 5.5.4.2 添加购物车

**接口**: `POST /cart/add`

```typescript
// server/api/cart/add.post.ts
import { requireAuth } from "~~/server/utils/auth";

export default defineEventHandler(async (event) => {
  // 使用JWT认证获取用户ID
  const { userId } = event.context.user;

  const body: CartAddRequest = await readBody(event);
  const { goods_id, sku_code, quantity = 1, sku_value } = body;

  // 检查库存
  const stockResult = await mySql`
    SELECT stock FROM skus WHERE goods_id = ${goods_id} AND sku_code = ${sku_code}
  `;

  if (stockResult[0].stock < quantity) {
    throw createError({ statusCode: 400, message: "库存不足" });
  }

  // 检查是否已存在（基于user_id和sku_code的唯一约束）
  const existingCartItem = await mySql`
    SELECT id, quantity FROM cart_items 
    WHERE user_id = ${userId} AND sku_code = ${sku_code}
  `;

  let cart_id: number;
  if (existingCartItem.length > 0) {
    // 更新数量
    cart_id = existingCartItem[0].id;
    await mySql`
      UPDATE cart_items 
      SET quantity = quantity + ${quantity}, updated_at = NOW()
      WHERE id = ${cart_id}
    `;
  } else {
    // 插入新记录
    const insertResult = await mySql`
      INSERT INTO cart_items (user_id, goods_id, sku_code, sku_value, quantity)
      VALUES (${userId}, ${goods_id}, ${sku_code}, ${sku_value}, ${quantity})
      RETURNING id
    `;
    cart_id = insertResult[0].id;
  }

  return { code: 200, msg: "success", data: { cart_id } };
});
```

### 5.5.4.3 获取购物车列表

**接口**: `POST /cart/list`

```typescript
// server/api/cart/list.post.ts
import { requireAuth } from "~~/server/utils/auth";

export default defineEventHandler(async (event) => {
  // 使用JWT认证获取用户ID
  const { userId } = event.context.user;
  const { page = 1, page_size = 10 } = await readBody(event);

  const result = await mySql`
    SELECT 
      c.id,
      c.goods_id::BIGINT AS goods_id,
      g.goods_name,
      g.shop_name,
      g.shop_id,
      g.image,
      s.price,
      c.quantity,
      s.stock,
      c.sku_code,
      c.sku_value,
      c.created_at,
      c.updated_at
    FROM cart_items c
    INNER JOIN goods g ON c.goods_id = g.id
    INNER JOIN skus s ON c.sku_code = s.sku_code
    WHERE c.user_id = ${userId}
    ORDER BY c.created_at DESC
    LIMIT ${page_size} OFFSET ${(page - 1) * page_size}
  `;

  return {
    code: 200,
    msg: "success",
    data: { CartList: result },
  };
});
```

### 5.5.4.4 更新购物车数量

**接口**: `POST /cart/update`

```typescript
// server/api/cart/update.post.ts
import { requireAuth } from "~~/server/utils/auth";

export default defineEventHandler(async (event) => {
  // 使用JWT认证获取用户ID
  const { userId } = event.context.user;

  const body: CartUpdateRequest = await readBody(event);
  const { cart_id, quantity } = body;

  // 参数验证
  if (!cart_id || quantity < 0) {
    throw createError({
      statusCode: 400,
      message: "购物车ID不能为空，数量不能为负数",
    });
  }

  // 检查购物车项是否存在且属于当前用户
  const cartItem = await mySql`
    SELECT c.id, c.goods_id, c.quantity, c.sku_code, g.price, s.stock
    FROM cart_items c
    INNER JOIN goods g ON c.goods_id = g.id
    INNER JOIN skus s ON c.sku_code = s.sku_code
    WHERE c.id = ${cart_id} AND c.user_id = ${userId}
  `;

  if (cartItem.length === 0) {
    throw createError({
      statusCode: 404,
      message: "购物车项不存在或无权限",
    });
  }

  // 如果数量为0，删除该购物车项
  if (quantity === 0) {
    await mySql`DELETE FROM cart_items WHERE id = ${cart_id}`;
    return {
      code: 200,
      msg: "购物车项已删除",
      data: { cart_id, new_quantity: 0 },
    };
  }

  // 更新购物车数量
  await mySql`
    UPDATE cart_items 
    SET quantity = ${quantity}, updated_at = NOW()
    WHERE id = ${cart_id}
  `;

  return {
    code: 200,
    msg: "更新成功",
    data: { cart_id, new_quantity: quantity },
  };
});
```

### 5.5.4.5 删除购物车商品

**接口**: `POST /cart/remove`

```typescript
// server/api/cart/remove.post.ts
import { requireAuth } from "~~/server/utils/auth";

export default defineEventHandler(async (event) => {
  // 使用JWT认证获取用户ID
  const { userId } = event.context.user;

  const body: CartDeleteRequest = await readBody(event);
  const { cart_ids } = body;

  // 参数校验
  if (!cart_ids || !Array.isArray(cart_ids) || cart_ids.length === 0) {
    throw createError({ statusCode: 400, message: "购物车ID数组不能为空" });
  }

  // 执行删除（自动过滤不属于当前用户的项）
  const deleteResult = await mySql`
    DELETE FROM cart_items
    WHERE id = ANY(${cart_ids})
    AND user_id = ${userId}
    RETURNING id
  `;

  return {
    code: 200,
    msg: "删除成功",
    data: { deleted_count: deleteResult.length },
  };
});
```

## 5.5.5 前端状态管理

### Pinia Store 核心逻辑

```typescript
// app/stores/cart.ts
export const useCartStore = defineStore("cart", () => {
  const cartList = ref<ShopItem[]>([]);
  const buyNowItem = ref<GoodsItem | null>(null);

  // 选中的商品
  const selectedItems = computed(() =>
    cartList.value
      .flatMap((item) => item.items)
      .filter((item) => item.selected),
  );

  // 选中商品总价
  const totalPrice = computed(() => {
    return selectedItems.value.reduce(
      (sum, item) => sum + item.quantity * parseFloat(item.price),
      0,
    );
  });

  // 是否全选
  const isAllSelected = computed(
    () =>
      cartList.value.length > 0 &&
      cartList.value
        .flatMap((item) => item.items)
        .every((item) => item.selected),
  );

  // 添加购物车
  function addCart(item: AddCartParams, isBuyNow = false) {
    if (isBuyNow) return;
    addCartApi(item).then((res) => {
      if (res.code === 200) {
        $message.success("已加入购物车");
      }
    });
  }

  // 删除购物车项
  function removeCart(cartIds: number[]) {
    removeCartApi(cartIds).then((res) => {
      if (res.code === 200) {
        loadCartList();
        $message.success("已从购物车移除");
      }
    });
  }

  // 加载购物车列表（按店铺分组）
  async function loadCartList() {
    const res = await getCartListApi({ page: 1, page_size: 100 });
    if (res.code === 200 && res.data.CartList) {
      const shopMap: Record<string, ShopItem> = {};
      res.data.CartList.forEach((item: GoodsItem) => {
        if (!shopMap[item.shop_name]) {
          shopMap[item.shop_name] = { shop_name: item.shop_name, items: [] };
        }
        shopMap[item.shop_name]!.items.push(item);
      });
      cartList.value = Object.values(shopMap);
    }
  }

  return {
    cartList,
    selectedItems,
    buyNowItem,
    totalPrice,
    isAllSelected,
    addCart,
    removeCart,
    loadCartList,
    toggleSelect,
    toggleSelectAll,
  };
});
```

## 5.5.8 关键设计要点

### 1. 数据分组策略

购物车数据按**店铺**分组展示，提升用户体验：

```typescript
const shopMap: Record<string, ShopItem> = {};
rawList.forEach((item) => {
  if (!shopMap[item.shop_name]) {
    shopMap[item.shop_name] = { shop_name: item.shop_name, items: [] };
  }
  shopMap[item.shop_name]!.items.push(item);
});
```

### 2. 状态同步机制

- 使用 Pinia 管理全局状态，确保各组件数据一致
- `onActivated` 钩子确保页面重新激活时刷新数据
- 选中状态存储在前端，提交订单时才传递给后端

### 3. 库存校验

添加购物车时双重校验：

- **添加前**：检查库存是否充足
- **结算前**：再次校验，防止超卖

### 4. 性能优化

- 使用 `computed` 计算属性缓存价格计算结果
- 分页加载购物车列表，避免一次性加载过多数据
- 本地状态管理减少不必要的 API 请求
